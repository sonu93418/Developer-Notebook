What is Node.js?

Node.js is a JavaScript runtime that allows you to run JavaScript outside the browser. Built on Chrome's V8 JavaScript engine, Node.js enables developers to use the same language for both frontend and backend development.

Key Features

  • Event-Driven: Built on event-driven, non-blocking I/O model
  • npm: Powerful package manager with millions of libraries
  • Lightweight: Efficient for I/O-heavy applications
  • Single-threaded: Event loop handles asynchronous operations
  • Scalable: Can handle thousands of concurrent connections

Getting Started

// Create a simple HTTP server
const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, {'Content-Type': 'text/plain'});
  res.end('Hello, World!\n');
});

server.listen(3000, () => {
  console.log('Server running on port 3000');
});

Express.js Framework

Express is the most popular Node.js framework for building web applications:

const express = require('express');
const app = express();

// Middleware
app.use(express.json());

// Routes
app.get('/', (req, res) => {
  res.json({ message: 'Hello, World!' });
});

app.post('/api/users', (req, res) => {
  const user = req.body;
  res.status(201).json(user);
});

app.listen(3000);

Async/Await Pattern

// Async function with await
async function fetchData() {
  try {
    const response = await fetch('https://api.example.com/data');
    const data = await response.json();
    return data;
  } catch (error) {
    console.error('Error:', error);
  }
}

// Using async in Express
app.get('/api/data', async (req, res) => {
  const data = await fetchData();
  res.json(data);
});

Database Integration

// MongoDB with Mongoose
const mongoose = require('mongoose');

const userSchema = new mongoose.Schema({
  name: String,
  email: { type: String, unique: true },
  createdAt: { type: Date, default: Date.now }
});

const User = mongoose.model('User', userSchema);

// PostgreSQL with pg
const { Pool } = require('pg');
const pool = new Pool({
  connectionString: process.env.DATABASE_URL
});

pool.query('SELECT * FROM users', (err, result) => {
  if (err) console.error(err);
  else console.log(result.rows);
});

RESTful API Best Practices

// Proper REST API structure
app.get('/api/users/:id', async (req, res) => {
  try {
    const user = await User.findById(req.params.id);
    if (!user) {
      return res.status(404).json({ error: 'User not found' });
    }
    res.json(user);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Middleware for authentication
const authenticate = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'Unauthorized' });
  // Verify token...
  next();
};

app.delete('/api/users/:id', authenticate, async (req, res) => {
  // Delete endpoint
});

File System Operations

// Reading and writing files
const fs = require('fs').promises;

async function readFile(path) {
  const content = await fs.readFile(path, 'utf-8');
  return content;
}

async function writeFile(path, data) {
  await fs.writeFile(path, JSON.stringify(data, null, 2));
}

Best Practices

  • Keep functions small and focused

  • Implement logging for debugging and monitoring
